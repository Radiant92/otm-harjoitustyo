90      public void playGame(int size) {
91          loser = false;
92          winner = false;
93          Label turns = new Label("" + newGame.getTurns());
94          Label line = new Label("Line ships left: " + newGame.getLineShips());
95          Label lShips = new Label("L-shaped ships left: " + newGame.getLShips());
96          Label uShips = new Label("U-shaped ships left: " + newGame.getUShips());
97          HBox labels = new HBox();
98          labels.setSpacing(20);
99          labels.getChildren().addAll(turns, line, lShips, uShips);
100         game.setTop(labels);
101 
102         for (int x = 0; x < size; x++) {
103             for (int y = 0; y < size; y++) {
104                 Button b = new Button(x + " - " + y);
105                 b.setPrefSize(50, 50);
106                 b.setOnMouseClicked(event -> {
107                     if (loser == true) {
108                         b.setText("LOSER");
109                     } else if (winner == true) {
110 
111                         b.setText("WIN");
112                     } else {
113                         String location = b.getText();
114                         if (!location.equals("O") && !location.equals("X")) {
115                             String[] split = location.split("-");
116 
117                             int buttonX = Integer.parseInt(split[0].trim());
118 
119                             int buttonY = Integer.parseInt(split[1].trim());
120                             String result = newGame.hit(buttonX, buttonY);
121 
122                             if (result.equals("miss")) {
123                                 b.setText("O");
124                                 turns.setText("" + newGame.getTurns());
125                                 if (newGame.didYouLose() == true) {
126                                     loser = true;
127                                 }
128                             } else if (result.equals("strike")) {
129                                 b.setText("X");
130                                 if (newGame.didYouWin() == true) {
131                                     winner = true;
132                                 }
133                             }
134                             line.setText("Line ships left: " + newGame.getLineShips());
135                             lShips.setText("L-shaped ships left: " + newGame.getLShips());
136                             uShips.setText("U-shaped ships left: " + newGame.getUShips());
137                         }
138                     }
139                 });
140                 board.add(b, x, y);
141 
142             }
143         }
144 
145     }
