240     public void playGame(int size) {
241 
242         Label turns = new Label("" + newGame.getTurns());
243         Label line = new Label("Line ships left: " + newGame.getLineShips());
244         Label lShips = new Label("L-shaped ships left: " + newGame.getLShips());
245         Label uShips = new Label("U-shaped ships left: " + newGame.getUShips());
246         HBox labels = new HBox();
247         labels.setSpacing(20);
248         labels.getChildren().addAll(turns, line, lShips, uShips);
249         game.setTop(labels);
250 
251         for (int x = 0; x < size; x++) {
252             for (int y = 0; y < size; y++) {
253                 Button b = new Button(x + " - " + y);
254                 b.setPrefSize(50, 50);
255                 b.setOnMouseClicked(event -> {
256 
257                     String location = b.getText();
258                     if (!location.equals("O") && !location.equals("X")) {
259                         String[] split = location.split("-");
260 
261                         int buttonX = Integer.parseInt(split[0].trim());
262 
263                         int buttonY = Integer.parseInt(split[1].trim());
264                         String result = newGame.hit(buttonX, buttonY);
265 
266                         if (result.equals("miss")) {
267                             b.setText("O");
268                             turns.setText("" + newGame.getTurns());
269                             if (newGame.didYouLose() == true) {
270 
271                                 goToMenu();
272 
273                             }
274                         } else if (result.equals("strike")) {
275                             b.setText("X");
276                             if (newGame.didYouWin() == true) {
277                                 if (size == 6) {
278                                     this.user.winEasy();
279                                 } else if (size == 8) {
280                                     this.user.winNormal();
281                                 } else {
282                                     this.user.winHard();
283                                 }
284                                 try {
285                                     this.database.save();
286                                 } catch (Exception ex) {
287 
288                                 }
289                                 goToMenu();
290 
291                             }
292                         }
293                         line.setText("Line ships left: " + newGame.getLineShips());
294                         lShips.setText("L-shaped ships left: " + newGame.getLShips());
295                         uShips.setText("U-shaped ships left: " + newGame.getUShips());
296                     }
297 
298                 });
299                 board.add(b, x, y);
300 
301             }
302         }
303 
304     }
