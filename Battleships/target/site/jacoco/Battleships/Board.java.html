package Battleships;


/**
 *
 * @author strohm
 */
public class Board {

    private int[][] board;

    public Board(int size) {
        this.board = new int[size][size];
        for (int y = 0; y < size; y++) {
            for (int x = 0; x < size; x++) {
                this.board[y][x] = 0;
          
            }
        }
    }
    public int[][] getBoard(){
        return this.board;
    }
}
